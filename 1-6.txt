#include <iostream>
#include <string>

using namespace std;

int main()
{
   int n;
    int a[10][10];
    cin>>n;
    for (int i =0;i<n;i++){
        for (int j = 0;j<n;j++){
            a[i][j]=0;
        }
    } 
    a[0][0] = 1;
    a[0][n-1] = 2;
    a[n-1][0] = 2;
    a[n-1][n-1] = 1;
    for (int i =0;i<n;i++){
        for (int j = 0;j<n;j++){
            cout<<a[i][j]<<" ";
        }
        cout<<"\n";
    }
}
